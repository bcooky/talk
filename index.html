/* style.css */ body { font-family: Arial, sans-serif; display: flex; flex-direction: column; align-items: center; justify-content: center; height: 100vh; background-color: #f4f4f4; } #chat-container { width: 300px; height: 400px; border: 1px solid #ccc; display: flex; flex-direction: column; padding: 10px; background-color: white; } #chat-box { flex: 1; overflow-y: auto; border-bottom: 1px solid #ccc; padding: 5px; } #message-input { width: 100%; padding: 5px; } button { margin-top: 5px; padding: 10px; cursor: pointer; } /* script.js */ const chatBox = document.getElementById("chat-box"); const messageInput = document.getElementById("message-input"); const sendButton = document.getElementById("send-button"); const callButton = document.getElementById("call-button"); const endCallButton = document.getElementById("end-call-button"); sendButton.addEventListener("click", () => { const message = messageInput.value; if (message.trim() !== "") { const messageElement = document.createElement("div"); messageElement.textContent = message; chatBox.appendChild(messageElement); messageInput.value = ""; } }); callButton.addEventListener("click", () => { alert("Starting call..."); callButton.disabled = true; endCallButton.disabled = false; }); endCallButton.addEventListener("click", () => { alert("Call ended."); callButton.disabled = false; endCallButton.disabled = true; });
